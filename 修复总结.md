private async replaceSelection(context: ActionContext, newContent: string): Promise<void> {
    const currentText = editor.getRange(context.selectionFrom, context.selectionTo);

    if (currentText !== context.selectedText) {
        const fullContent = editor.getValue();
        const originalIndex = fullContent.indexOf(context.selectedText);

        if (originalIndex === -1) {
            throw new Error('原选区内容已被修改或删除，无法替换。请撤销更改后重试。');
        }

        const originalPos = editor.posToOffset(context.selectionFrom);
        const foundPos = originalIndex;

        if (Math.abs(foundPos - originalPos) > 10) {
            throw new Error(`检测到内容位置变化（偏移 ${Math.abs(foundPos - originalPos)} 字符），为安全起见已取消操作。`);

### 8. 错误处理改进
**位置：** `src/services/ai.ts`

**修复内容：**
- ✅ 添加网络错误自动重试机制（最多3次）
- ✅ 指数退避策略（1秒、2秒、4秒）
- ✅ 区分可重试和不可重试的错误
- ✅ 改进超时处理

**代码示例：**
```typescript
export class AIService {
    private static readonly MAX_RETRIES = 3;
    private static readonly RETRY_DELAY_BASE = 1000;

    async chat(prompt: string, content: string): Promise<AIResponse> {
        for (let attempt = 1; attempt <= AIService.MAX_RETRIES; attempt++) {
            try {
                const result = await this.makeRequest(prompt, content);
                return result;
            } catch (error) {
                if (attempt < AIService.MAX_RETRIES && this.isRetryableError(error)) {
                    const delay = Math.min(
                        AIService.RETRY_DELAY_BASE * Math.pow(2, attempt - 1),
                        10000
                    );
                    console.log(`重试 ${attempt}/${AIService.MAX_RETRIES}，等待 ${delay}ms`);
                    await new Promise(resolve => setTimeout(resolve, delay));
                    continue;
                }
                // 返回错误
            }
        }
    }

    private isRetryableError(error: any): boolean {
        // 超时、网络错误、5xx错误、429错误可重试
        return error.isTimeout ||
               error instanceof TypeError ||
               (error.status >= 500 && error.status < 600) ||
               error.status === 429;
    }
}
```
